Listing etudiant -> cours : 

SELECT etudiant.Prenom, etudiant.Nom, assister.NomCours, journee.Jour, assister.HeureDebut, assister.HeureFin, cours.NomLocal, professeur.Prenom, professeur.Nom FROM etudiant INNER JOIN assister ON assister.AdresseMail = etudiant.AdresseMail LEFT JOIN journee ON assister.IdJournee = journee.IdJournee INNER JOIN composer ON journee.IdJournee = composer.IdJournee LEFT JOIN cours ON composer.NomCours = cours.NomCours AND composer.HeureDebut = cours.HeureDebut AND composer.HeureFin = cours.HeureFin AND composer.IdProfesseur = cours.IdProfesseur INNER JOIN professeur ON cours.IdProfesseur = professeur.IdProfesseur WHERE etudiant.validation = 1 GROUP BY etudiant.AdresseMail, assister.IdJournee, assister.NomCours, assister.HeureDebut, assister.HeureFin

Programme individuel : 

SELECT etudiant.Nom, etudiant.Prenom, assister.Adressemail, journee.Jour, assister.NomCours, assister.HeureDebut, assister.HeureFin, professeur.Nom, professeur.Prenom FROM etudiant INNER JOIN assister ON assister.AdresseMail = etudiant.AdresseMail LEFT JOIN journee ON assister.IdJournee = journee.IdJournee INNER JOIN composer ON journee.IdJournee = composer.IdJournee LEFT JOIN cours ON composer.NomCours = cours.NomCours INNER JOIN professeur ON cours.IdProfesseur = professeur.IdProfesseur WHERE etudiant.AdresseMail = 'regis3d@hotmail.com' AND composer.HeureDebut = cours.HeureDebut AND composer.HeureFin = cours.HeureFin AND composer.IdProfesseur = cours.IdProfesseur GROUP BY etudiant.AdresseMail, assister.IdJournee, assister.NomCours, assister.HeureDebut, assister.HeureFin, assister.IdProfesseur ORDER BY assister.IdJournee, assister.HeureDebut
